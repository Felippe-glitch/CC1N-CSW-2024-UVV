section .data
        hello db 'Hello, World!',0x0A ; A string que será exibida seguida por um caractere de nova linha
    
    section .text
        global _start ; A entrada principal para o programa
    
    _start:
        ; Sistema de chamada para escrever na saída padrão
        mov eax, 4         ; Número da chamada do sistema para sys_write (4)
        mov ebx, 1         ; Descritor de arquivo para a saída padrão (stdout)
        mov ecx, hello     ; Endereço da string para exibir
        mov edx, 14        ; Comprimento da string (13 caracteres + 1 para a nova linha)
        int 0x80           ; Interrupção para chamar o kernel
    
        ; Sistema de chamada para sair do programa
        mov eax, 1         ; Número da chamada do sistema para sys_exit (1)
        xor ebx, ebx       ; Código de saída 0
        int 0x80           ; Interrupção para chamar o kernel
